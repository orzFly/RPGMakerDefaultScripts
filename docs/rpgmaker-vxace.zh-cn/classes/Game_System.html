def on_before_save
  @save_count += 1
  @version_id = $data_system.version_id
  @frames_on_save = Graphics.frame_count
  @bgm_on_save = RPG::BGM.last
  @bgs_on_save = RPG::BGS.last
end

def playtime_s
  hour = playtime / 60 / 60
  min = playtime / 60 % 60
  sec = playtime % 60
  sprintf("%02d:%02d:%02d", hour, min, sec)
end
