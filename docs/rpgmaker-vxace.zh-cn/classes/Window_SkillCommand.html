def make_command_list
  return unless @actor
  @actor.added_skill_types.sort.each do |stype_id|
    name = $data_system.skill_types[stype_id]
    add_command(name, :skill, true, stype_id)
  end
end

def select_last
  skill = @actor.last_skill.object
  if skill
    select_ext(skill.stype_id)
  else
    select(0)
  end
end
