def draw_item(index)
  return unless @actor
  rect = item_rect_for_text(index)
  change_color(system_color, enable?(index))
  draw_text(rect.x, rect.y, 92, line_height, slot_name(index))
  draw_item_name(@actor.equips[index], rect.x + 92, rect.y, enable?(index))
end
