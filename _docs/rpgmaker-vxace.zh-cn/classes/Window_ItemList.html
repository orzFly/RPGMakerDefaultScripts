def draw_item(index)
  item = @data[index]
  if item
    rect = item_rect(index)
    rect.width -= 4
    draw_item_name(item, rect.x, rect.y, enable?(item))
    draw_item_number(rect, item)
  end
end

# File Window_ItemList.rb, line 57
def include?(item)
  case @category
  when :item
    item.is_a?(RPG::Item) && !item.key_item?
  when :weapon
    item.is_a?(RPG::Weapon)
  when :armor
    item.is_a?(RPG::Armor)
  when :key_item
    item.is_a?(RPG::Item) && item.key_item?
  else
    false
  end
end
